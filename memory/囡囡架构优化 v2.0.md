## 🔧 优化 1：ReAct 决策逻辑

### 优化前
```
主人问题 → 意图识别 → 工具调用 → 回复
```

### 优化后（ReAct 范式）
```
主人问题 → Thought(思考) → Action(行动) → Observation(观察) → 
           检查完成？→ 否 → 继续循环
                      → 是 → 生成回复
```

### ReAct 实现
```python
class ReActEngine:
    """ReAct 推理引擎"""
    
    def __init__(self, max_steps=5):
        self.max_steps = max_steps
        self.thoughts = []
    
    def reason(self, query, context):
        """ReAct 推理循环"""
        self.thoughts = []
        
        for step in range(self.max_steps):
            # Thought: 思考当前状态
            thought = self.generate_thought(query, context, step)
            self.thoughts.append(thought)
            
            # Action: 决定行动
            action = self.decide_action(thought)
            
            # Observation: 执行并观察结果
            observation = self.execute_action(action)
            
            # 检查是否完成
            if self.is_complete(observation, thought):
                break
            
            # 更新上下文
            context['observations'].append(observation)
        
        # 生成最终回复
        return self.generate_response(query, self.thoughts, observation)
    
    def generate_thought(self, query, context, step):
        """生成思考"""
        prompt = f"""
思考步骤 {step + 1}:
问题：{query}
已知信息：{context}

我需要：
1. 分析问题的核心需求
2. 检查已有的信息和工具
3. 决定下一步行动

思考：
"""
        return llm.generate(prompt)
    
    def decide_action(self, thought):
        """决定行动"""
        # 基于思考选择工具
        tools = ['memory_search', 'web_search', 'web_fetch', 'exec', ...]
        return select_best_tool(thought, tools)
    
    def generate_response(self, query, thoughts, final_observation):
        """生成最终回复"""
        prompt = f"""
基于以下推理过程生成回复：

问题：{query}

推理过程：
{chr(10).join([f"步骤{i+1}: {t}" for i, t in enumerate(thoughts)])}

最终观察：{final_observation}

请生成温暖、详细、结构化的回复：
"""
        return llm.generate(prompt)
```

### 应用示例
```

步骤 2 Action: memory_search("AI 落地 场景")

步骤 3 Thought: 已收集足够信息，可以生成建议
步骤 3 Action: generate_response
```

```python
class MemoryManager:
    """记忆管理器（优化版）"""
    
    def __init__(self):
        self.short_term = []  # 短期记忆（工作记忆）
        self.long_term = {}   # 长期记忆（向量化）
        self.importance_threshold = 0.7
    
    def add_memory(self, content, importance=None):
        """添加记忆"""
        # 计算重要性评分
        if importance is None:
            importance = self.calculate_importance(content)
        
        memory = {
            'content': content,
            'importance': importance,
            'created_at': datetime.now(),
            'access_count': 0,
            'last_accessed': None
        }
        
        # 短期记忆
        self.short_term.append(memory)
        
        # 超过阈值→转入长期记忆
        if importance >= self.importance_threshold:
            self.consolidate(memory)
    
    def calculate_importance(self, content):
        """计算记忆重要性"""
        # 因素 1：主人明确表达
        explicit_keywords = ['记住', '偏好', '喜欢', '重要']
        explicit_score = 0.3 if any(k in content for k in explicit_keywords) else 0
        
        # 因素 2：重复提及
        repetition_score = self.count_mentions(content) * 0.2
        
        # 因素 3：情感强度
        emotion_score = self.analyze_emotion(content) * 0.2
        
        # 因素 4：实用性
        utility_score = self.assess_utility(content) * 0.3
        
        return explicit_score + repetition_score + emotion_score + utility_score
    
    def consolidate(self, memory):
        """记忆巩固（短期→长期）"""
        # 向量化
        vector = generate_embedding(memory['content'])
        self.long_term[memory['content']] = {
            **memory,
            'vector': vector
        }
    
    def forget(self):
        """遗忘机制（Ebbinghaus 遗忘曲线）"""
        now = datetime.now()
        to_remove = []
        
        for mem in self.short_term:
            # 计算遗忘概率
            age_hours = (now - mem['created_at']).total_seconds() / 3600
            forget_probability = 1 - math.exp(-age_hours / 24)  # 24 小时半衰期
            
            # 重要性低的更容易遗忘
            if random.random() < forget_probability * (1 - mem['importance']):
                to_remove.append(mem)
        
        # 清理
        for mem in to_remove:
            self.short_term.remove(mem)
```

```python
class ContextEngine:
    """上下文引擎（优化版）"""
    
    def __init__(self):
        self.conversation_history = []  # 对话历史
        self.topic_sequence = []  # 话题序列
        self.user_state = {}  # 用户状态
    
    def update_context(self, query, response):
        """更新上下文"""
        # 记录对话
        self.conversation_history.append({
            'query': query,
            'response': response,
            'timestamp': datetime.now()
        })
        
        # 提取话题
        topic = self.extract_topic(query)
        self.topic_sequence.append({
            'topic': topic,
            'timestamp': datetime.now()
        })
        
        # 更新用户状态
        self.update_user_state(query)
    
    def get_relevant_context(self, query, top_k=3):
        """获取相关上下文"""
        # 向量化查询
        query_vector = generate_embedding(query)
        
        # 检索历史对话
        relevant = []
        for conv in self.conversation_history[-20:]:  # 最近 20 条
            conv_vector = generate_embedding(conv['query'])
            sim = cosine_similarity(query_vector, conv_vector)
            if sim > 0.5:
                relevant.append((conv, sim))
        
        # 排序取 Top-K
        relevant.sort(key=lambda x: x[1], reverse=True)
        return [r[0] for r in relevant[:top_k]]
    
    def predict_next_topic(self):
        """预测下一个话题"""
        if len(self.topic_sequence) < 3:
            return None
        
        # 分析话题序列模式

                'predicted': topics[-1],
```

```python
class NannanEvaluator:
    """年年质量评估器"""
    
    def __init__(self):
        self.metrics = {
            'task_completion_rate': 0,  # 任务完成率
            'response_time': 0,  # 响应时间
            'memory_accuracy': 0,  # 记忆检索准确率
            'prediction_accuracy': 0,  # 意图预测准确率
            'user_satisfaction': 0,  # 用户满意度
            'cost_savings': 0  # 成本节省
        }
        self.history = []
    
    def evaluate(self, query, response, start_time, end_time):
        """单次评估"""
        evaluation = {
            'timestamp': datetime.now(),
            'query': query,
            'response': response,
            'response_time': (end_time - start_time).total_seconds(),
            'task_completed': self.check_task_completion(response),
            'memory_used': self.check_memory_used(response),
            'prediction_correct': self.check_prediction(response),
        }
        
        self.history.append(evaluation)
        self.update_metrics()
    
    def update_metrics(self):
        """更新指标"""
        recent = self.history[-100:]  # 最近 100 次
        
        # 任务完成率
        completed = sum(1 for h in recent if h['task_completed'])
        self.metrics['task_completion_rate'] = completed / len(recent)
        
        # 平均响应时间
        self.metrics['response_time'] = sum(h['response_time'] for h in recent) / len(recent)
        
        # 记忆准确率
        memory_correct = sum(1 for h in recent if h['memory_used'])
        self.metrics['memory_accuracy'] = memory_correct / len(recent)
        
        # 意图预测准确率
        prediction_correct = sum(1 for h in recent if h['prediction_correct'])
        self.metrics['prediction_accuracy'] = prediction_correct / len(recent)
    
    def get_report(self):
        """生成评估报告"""
        return f"""
📊 年年质量评估报告
━━━━━━━━━━━━━━━━━━━━
✅ 任务完成率：{self.metrics['task_completion_rate']:.1%} (目标：>90%)
⚡ 响应时间：{self.metrics['response_time']:.2f}s (目标：<2s)
🧠 记忆准确率：{self.metrics['memory_accuracy']:.1%} (目标：>85%)
🎯 预测准确率：{self.metrics['prediction_accuracy']:.1%} (目标：>70%)
💰 成本节省：{self.metrics['cost_savings']:.1%} (目标：>60%)
━━━━━━━━━━━━━━━━━━━━
评估次数：{len(self.history)}
"""
```
