### 3. 意图预测

```python
class IntentPredictor:
    def predict(self, patterns, interest_model):
        predictions = []
        
        # 模式 1：连续学习 → 可能需要深入或休息
        if patterns['continuous_learning']:
            predictions.append({
                'type': 'continue_or_rest',
                'confidence': 0.8,
                'suggestion': '继续深入 or 休息建议'
            })
        
        # 模式 2：学习完成 → 可能需要实践指导
        if patterns['theory_complete']:
            predictions.append({
                'type': 'implementation_help',
                'confidence': 0.7,
                'suggestion': '提供实践方案'
            })
        
        # 模式 3：对比学习 → 可能需要决策支持
        if patterns['comparison']:
            predictions.append({
                'type': 'decision_support',
                'confidence': 0.75,
                'suggestion': '提供对比分析表'
            })
        
        return predictions
```

### 4. 建议生成

```python
class SuggestionGenerator:
    def generate(self, predictions, context):
        suggestions = []
        
        for pred in predictions:
            if pred['type'] == 'continue_or_rest':
                suggestions.append({
                    'emoji': '☕',
                    'text': f'主人已学习 {context["duration"]} 小时，'
                           f'需要休息一下吗？或者年年继续讲解？',
                    'actions': ['继续', '休息', '总结']
                })
            
            elif pred['type'] == 'implementation_help':
                suggestions.append({
                    'emoji': '💡',
                    'text': f'年年已准备好 {context["topic"]} 的实施步骤，'
                           f'需要现在开始吗？',
                    'actions': ['开始', '查看方案', '稍后']
                })
            
            elif pred['type'] == 'decision_support':
                suggestions.append({
                    'emoji': '📊',
                    'text': f'年年整理了对比分析表，'
                           f'帮助主人做出决策～',
                    'actions': ['查看', '跳过']
                })
        
        return suggestions
```
