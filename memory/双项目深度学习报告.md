            action = self.select_tool(thought)
            actions.append(action)
            
            # Observation: 执行并观察
            observation = self.execute(action)
            observations.append(observation)
            
            # ✨ NEW: Reflection 反思
            if i > 0:
                reflection = self.reflect(thoughts, actions, observations)
                if reflection['should_adjust']:
                    # 调整策略
                    pass
            
            # 检查是否完成
            if self.is_complete(observation):
                break
        
        return self.generate_response(thoughts, actions, observations)
    
    def reflect(self, thoughts, actions, observations):
        """反思改进"""
        prompt = f"""
        回顾之前的推理过程：
