### 步骤 2：提取记忆

```python
# 伪代码
def extract_memories(conversations):
    new_memories = []
    
    for conv in conversations:
        # 应用 EXTRACTION_RULES.md
        memory = apply_extraction_rules(conv)
        
        if memory:
            # 检查是否重复
            if not is_duplicate(memory):
                # 创建记忆项
                create_memory_item(memory)
                new_memories.append(memory)
    
    return new_memories
```

    # 更新统计
    index.stats.total += len(new_memories)
    index.stats.today += len(new_memories)

    for memory in new_memories:

## 📝 话题热度追踪

### 数据结构

```json
{
  "topics": {
    "多 Agent 协同": {
      "count": 5,
      "last_mentioned": "2026-03-02 17:00",
      "trend": "up"
    },
    "记忆系统": {
      "count": 3,
      "last_mentioned": "2026-03-02 16:45",
      "trend": "stable"
    }
  },
  "last_updated": "2026-03-02 17:35"
}
```

### 热度计算

```python
def calculate_heat(topics, days=7):
    heat = {}
    for topic, data in topics.items():
        # 频率分数
        freq_score = data.count / days
        
        # 新鲜度分数
        recency = days_since(data.last_mentioned)
        recency_score = 1 / (recency + 1)
        
        # 趋势分数
        trend_score = 1.5 if data.trend == "up" else 1.0
        
        # 综合热度
        heat[topic] = freq_score * recency_score * trend_score
    
    return heat
```
